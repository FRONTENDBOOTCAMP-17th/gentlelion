## 3-1. `BASE_URL` 중복 제거 — `productDetailApi.js`를 `share-api`로 통합 (중요도: 높음)

### 문제

`packages/share-api/index.js`에 이미 `BASE_URL`이 정의되어 있는데, `packages/user/components/API/product/productDetailApi.js` 1번 줄에도 똑같은 `BASE_URL`이 따로 있어요. 서버 주소가 바뀌면 두 파일을 모두 수정해야 합니다.

### 해결 방법

**Step 1.** `packages/user/components/API/product/productDetailApi.js` 파일을 통째로 교체하세요.

```js
// 수정 전 (packages/user/components/API/product/productDetailApi.js)
const BASE_URL = "https://api.fullstackfamily.com/api/gentlelion/v1";

export async function fetchProduct(productId) {
  const res = await fetch(BASE_URL + "/products/" + productId);
  if (!res.ok) throw new Error("product_not_found");
  const json = await res.json();
  return json.data;
}

export async function fetchSimilarProducts(category, productId) {
  const res = await fetch(
    BASE_URL + "/products?category=" + category + "&limit=6",
  );
  if (!res.ok) return [];
  const json = await res.json();
  return (json.data || [])
    .filter(function (p) {
      return String(p.id) !== String(productId);
    })
    .slice(0, 5);
}
```

```js
// 수정 후
import { get } from "@gentlelion/share-api";

export async function fetchProduct(productId) {
  const json = await get(`/products/${productId}`);
  return json.data;
}

export async function fetchSimilarProducts(category, productId) {
  try {
    const json = await get(`/products?category=${category}&limit=6`);
    return (json.data || [])
      .filter((p) => String(p.id) !== String(productId))
      .slice(0, 5);
  } catch {
    return [];
  }
}
```

**Step 2.** 이 파일을 사용하는 곳에서 변경이 필요한지 확인하세요.
- `packages/user/pages/product-detail.html` 내부의 `<script>`
- `packages/user/components/product-detail/Productdetailrenderer.js`

호출 방식(`fetchProduct(productId)`)이 그대로라서 **호출부는 수정하지 않아도 됩니다.**

> **쉽게 말하면:** 공용 우산(`share-api`)을 만들어놓고 혼자 비 맞고 가는 파일이 있었어요. 우산 밑으로 들여보내는 작업이에요.

```js
export function createColorRow({ value = "", onChange, onRemove, container }) {
  const row = document.createElement("div");
  row.className = "flex items-center gap-2";

  const input = document.createElement("input");

  const svg = document.createElementNS("http://www.w3.org/2000/svg", "svg");
  svg.setAttribute("class", "w-5 h-5");
  svg.setAttribute("fill", "none");
  svg.setAttribute("stroke", "currentColor");
  svg.setAttribute("viewBox", "0 0 24 24");

  const path = document.createElementNS("http://www.w3.org/2000/svg", "path");
  path.setAttribute(
    "d",
    "M19 7l-.867 12.142A2 2 0 0116.138 21H7.862a2 2 0 01-1.995-1.858L5 7m5 4v6m4-4v6m1-10V4a1 1 0 00-1-1h-4a1 1 0 00-1 1v3M4 7h16",
  );
  svg.appendChild(path);
  removeBtn.appendChild(svg);

  removeBtn.addEventListener("click", () => {
    const rows = container.querySelectorAll(".flex.items-center.gap-2");
    if (rows.length === 1) {
      input.value = "";
      if (onChange) onChange();
      return;
    }
    row.remove();
    if (onRemove) onRemove();
  });

  input.addEventListener("input", () => {
    if (onChange) onChange();
  });

  row.appendChild(input);
  row.appendChild(removeBtn);

  return row;
}
```

`packages/user/components/order/orderSummation.js` 10~12줄에서 `container.style.maxHeight = "0px"` 같은 인라인 스타일을 JS에서 지정하고 있어요. 프로젝트 규칙(TailwindCSS 우선)에 어긋납니다.

### 해결 방법

**Step 1.** HTML(또는 `order.html`)의 해당 컨테이너에 Tailwind 클래스를 미리 넣어두세요.

```html
<!-- 수정 전 -->
<div class="order-summation-container">
  ...
</div>

<!-- 수정 후 -->
<div class="order-summation-container max-h-0 overflow-hidden transition-[max-height] duration-300 ease-out">
  ...
</div>
```

**Step 2.** `orderSummation.js`에서 `style.*` 지정 부분을 제거하고 `classList.toggle`로 바꿉니다.

```js
// 수정 전
buttons.forEach((button, index) => {
  const container = containers[index];
  if (!container) return;

  container.style.maxHeight = "0px";
  container.style.overflow = "hidden";
  container.style.transition = "max-height 0.3s ease";

  button.addEventListener("click", () => {
    const isOpen = container.style.maxHeight !== "0px";
    container.style.maxHeight = isOpen ? "0px" : container.scrollHeight + "px";
  });
});
```

```js
// 수정 후
buttons.forEach((button, index) => {
  const container = containers[index];
  if (!container) return;

  button.addEventListener("click", () => {
    const isOpen = !container.classList.contains("max-h-0");
    if (isOpen) {
      container.classList.add("max-h-0");
      container.classList.remove("max-h-[500px]");
    } else {
      container.classList.remove("max-h-0");
      container.classList.add("max-h-[500px]");
    }
  });
});
```

> **쉽게 말하면:** 옷 색깔을 옷가게(CSS, Tailwind)에서 고르는 게 아니라 입고 나와서 염색약을 뿌리는(JS `style.*`) 상황이었어요. 옷가게에서 입고 나오는 게 훨씬 자연스럽죠.

**Step 2.** `arrivals.js`에서 템플릿을 fetch해 복제하는 패턴으로 바꿉니다.

```js
// 수정 후 arrivals.js
import { getProductList } from "../API/product/productListApi.js";
import { getDisplayImage } from "../../utils/productImage.js";

export async function renderArrivals() {
  const wrapper = document.getElementById("arrivals-wrapper");
  const thumbsWrapper = document.getElementById("arrivals-thumbs");
  if (!wrapper || !thumbsWrapper) return;

  const res = await fetch("/components/slide/arrivalsSlide.html");
  if (!res.ok) return;
  const templateHtml = await res.text();
  const parser = new DOMParser();

  const result = await getProductList("collection", 10, 1);
  const products = result.data;

  wrapper.innerHTML = "";
  thumbsWrapper.querySelectorAll(".thumb-item").forEach((el) => el.remove());

  products.forEach((item, index) => {
    const doc = parser.parseFromString(templateHtml, "text/html");
    const slide = doc.body.firstElementChild;

    const link = slide.querySelector(".arrivals-link");
    if (link) link.href = `/pages/product-detail.html?productId=${item.id}`;

    const img = slide.querySelector(".arrivals-img");
    if (img) {
      img.src = getDisplayImage(item);
      img.alt = item.name ?? "";
    }

    const nameEl = slide.querySelector(".arrivals-name");
    if (nameEl) nameEl.textContent = item.name;

    const priceEl = slide.querySelector(".arrivals-price");
    if (priceEl) priceEl.textContent = "₩" + item.price.toLocaleString();

    wrapper.appendChild(slide);

    // 썸네일도 같은 방식으로…
  });
}
```

> **쉽게 말하면:** "문자열을 HTML로 해석"하는 `innerHTML` 방식은 편하지만 조심할 것들이 많아요. **미리 만들어둔 HTML 템플릿을 가져와서 빈칸만 채우는** 방식이 안전하고, 이 프로젝트에서 이미 `productCard.js`가 쓰고 있는 패턴이에요.

## 3-12. 이미지 업로드 병렬 처리 (중요도: 낮음)

### 문제

`packages/admin/pages/API/product/imageApi.js`가 파일을 하나씩 순차 업로드해요. 5개면 5번 기다려야 합니다.

### 해결 방법

**Step 1.** 함수를 `Promise.all` 기반으로 바꿉니다.

```js
// 수정 전
import { upload } from "@gentlelion/share-api";

export async function uploadImages(files) {
  const urls = [];
  for (const file of files) {
    const formData = new FormData();
    formData.append("file", file);
    const result = await upload("/admin/images", formData);
    urls.push(result.data.imageUrl);
  }
  return urls;
}
```

```js
// 수정 후
import { upload } from "@gentlelion/share-api";

export async function uploadImages(files) {
  const tasks = Array.from(files).map((file) => {
    const formData = new FormData();
    formData.append("file", file);
    return upload("/admin/images", formData);
  });
  const results = await Promise.all(tasks);
  return results.map((r) => r.data.imageUrl);
}
```

> **쉽게 말하면:** "한 사람씩 줄 서서 택배 보내기"에서 "동시에 5개 창구에서 보내기"로 바꾸는 거예요. 합계는 같지만 총 걸리는 시간이 훨씬 짧아져요.
